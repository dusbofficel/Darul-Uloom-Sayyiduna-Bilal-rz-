    // slider images (placeholders)

    const slides = ['img1.jpg','img2.jpg','img3.jpg','img4.jpg','img5.jpg'];

    const slidesContainer = document.getElementById('slides');

    const dotsContainer = document.getElementById('sliderDots');

    let idx = 0, interval = null;

 

    function buildSlider(lang='ur'){

      slidesContainer.innerHTML = '';

      dotsContainer.innerHTML = '';

      const caps = content[lang].sliderCaptions || [];

      slides.forEach((src,i)=>{

        const slide = document.createElement('div'); slide.className='slide';

        const img = document.createElement('img'); img.src=src; img.alt=slide ${i+1};

        const cap = document.createElement('div'); cap.className='caption'; cap.textContent = caps[i] || '';

        slide.appendChild(img); slide.appendChild(cap); slidesContainer.appendChild(slide);

        const dot = document.createElement('span'); dot.className='dot'; if(i===0) dot.classList.add('active');

        dot.onclick = ()=>{ goTo(i); resetAuto(); }; dotsContainer.appendChild(dot);

      });

      idx=0;

      slidesContainer.style.transform = 'translateX(0%)';

    }

    function next(){ idx=(idx+1)%slides.length; slidesContainer.style.transform=translateX(-${idx*100}%); updateDots(); }

    function goTo(i){ idx=i; slidesContainer.style.transform=translateX(-${idx*100}%); updateDots(); }

    function updateDots(){ const ds = dotsContainer.querySelectorAll('.dot'); ds.forEach(d=>d.classList.remove('active')); if(ds[idx]) ds[idx].classList.add('active'); }

    function startAuto(){ stopAuto(); interval=setInterval(next,3000); }

    function stopAuto(){ if(interval) clearInterval(interval); interval=null; }

    function resetAuto(){ stopAuto(); startAuto(); }

 

    // set language (global)
